## 1. 架构概览

### 1.1 三层存储架构

RikkaHub 采用 **三层分离** 的上下文存储架构：

```
┌─────────────────────────────────────────────────────┐
│              L1: 对话元数据层 (Metadata)             │
│         conversationentity + message_node            │
├─────────────────────────────────────────────────────┤
│              L2: P 层逐字数据 (Verbatim)             │
│    message_node_text + FTS4 + 倒排索引              │
├─────────────────────────────────────────────────────┤
│           L3: 归档压缩层 (Archive)                  │
│    archive_summary + vector_index + verbatim_artifact │
└─────────────────────────────────────────────────────┘
```

### 1.2 数据表清单

| 表名 | 层级 | 用途 | 记录数估算 |
|------|------|------|-----------|
| `conversationentity` | L1 | 对话元数据 | 1 |
| `message_node` | L1 | 消息节点 JSON | ~100 |
| `message_node_text` | L2 | 逐字文本（根源） | ~100 |
| `message_node_fts` | L2 | FTS4 全文搜索 | ~100 |
| `message_token_index` | L2 | 倒排索引（兜底） | ~10K |
| `verbatim_artifact` | L2 | 逐字素材库 | ~10 |
| `archive_summary` | L3 | 语义归档摘要 | ~5 |
| `vector_index` | L3 | 向量索引 | ~5 |

private suspend fun saveMessageNodes(conversationId: String, nodes: List<MessageNode>) {
    val entities = nodes.mapIndexed { index, node ->
        MessageNodeEntity(
            id = node.id.toString(),
            conversationId = conversationId,
            nodeIndex = index,
            messages = JsonInstant.encodeToString(node.messages),

    }
}

**构建 P 层数据** - VerbatimVaultService.kt:42-120

```kotlin
suspend fun buildMessageNodeText(
    nodeId: String,
    conversationId: String,
    nodeIndex: Int,
    messages: List<UIMessage>
) {
    if (messages.isEmpty()) return

    val firstMessage = messages.first()
    val roleInt = firstMessage.role.ordinal

    // 提取文本（从 parts 数组）
    val textBuilder = StringBuilder()
    for (part in firstMessage.parts) {
        if (part is UIMessagePart.Text) {
            textBuilder.append(part.text)
        }
    }

    val rawText = textBuilder.toString()
    if (rawText.isEmpty()) return

    // 归一化搜索文本（CJK 分词）
    val searchText = normalizeForSearch(rawText)

    // 插入 message_node_text（触发器自动同步 FTS）
    messageNodeTextDao.insertOrReplace(
        MessageNodeTextEntity(
            nodeId = nodeId,
            conversationId = conversationId,
            nodeIndex = nodeIndex,
            role = roleInt,
            rawText = rawText,
            searchText = searchText
        )
    )

    // 构建倒排索引（兜底方案）
    val tokens = searchText.split(" ").filter { it.isNotBlank() }.distinct().take(64)
    for (token in tokens) {
        messageNodeTextDao.insertTokenIndex(
            tokenId = "${conversationId}_${nodeIndex}_$token",
            token = token,
            conversationId = conversationId,
            nodeIndex = nodeIndex,
            nodeId = nodeId
        )
    }

    // 扫描并生成 verbatim_artifact（写死规则）
    if (roleInt == 0) {  // 只处理用户消息
        detectAndCreateArtifact(...)
    }
}
```

### 6.2 核心代码证据

**删除方法** - ConversationRepository.kt:156-189

```kotlin
suspend fun deleteConversation(conversation: Conversation) {
    val conversationIdStr = conversation.id.toString()

    database.withTransaction {
        // 手动清理关联表（不依赖外键 CASCADE）

        // 1. 删除逐字素材（verbatim_artifact）
        verbatimVaultService.deleteArtifactsByConversation(conversationIdStr)

        // 2. 删除 P 层文本（message_node_text + FTS + 倒排索引）
        verbatimVaultService.deleteMessageNodeTextByConversation(conversationIdStr)

        // 3. 删除向量索引（通过 archive_id）
        val archiveSummaries = archiveSummaryDao.getListByConversationId(conversationIdStr)
        archiveSummaries.forEach { archive ->
            vectorIndexDao.deleteByArchiveId(archive.id)
        }

        // 4. 删除归档摘要（archive_summary）
        archiveSummaryDao.deleteByConversationId(conversationIdStr)

        // 5. 删除会话（message_node 会通过 CASCADE 自动删除）
        conversationDAO.delete(
            conversationToConversationEntity(conversation)
        )
    }

    // 清理文件系统
    context.deleteChatFiles(fullConversation.files)
}
```

### 9.2 FTS DELETE 触发器崩溃

**问题**: 删除对话时报 `SQLiteException: SQL logic error (code 1 SQLITE_ERROR)`

**根本原因**: FTS4 的 DELETE 触发器在 SQLite 3.32.2 上不稳定

1. 移除 DELETE 和 UPDATE 触发器
